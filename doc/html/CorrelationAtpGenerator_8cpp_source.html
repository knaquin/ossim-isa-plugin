75 bool CorrelationAtpGenerator::generateTiePointList(TiePointList& ctpList)
76 {
77  initialize();
78 
79  if (!m_refChain || !m_cmpChain)
80  return false;
81 
82  shared_ptr<ossimImageData> ref_patch;
83  shared_ptr<ossimImageData> cmp_patch;
84  shared_ptr<CorrelationTiePoint> ctp = 0;
85  TiePointList tileCTPs;
86  int tpId = 1;
87 
88  // Tile based to handle large overlaps:
89  shared_ptr<ossimImageSourceSequencer> sequencer (new ossimImageSourceSequencer);
90  sequencer->connectMyInputTo(m_refChain.get());
91  sequencer->setAreaOfInterest(m_aoiView);
92 
93  // Need to search for features in the REF tile first, then consider only those locations
94  // for correlating:
95  shared_ptr<ossimImageData> ref_tile (sequencer->getNextTile().get());
96  while (ref_tile)
97  {
98  tileCTPs.clear();
99  if (ref_tile->getDataObjectStatus() == OSSIM_EMPTY)
100  return false;
101  vector<ossimIpt> featurePoints;
102  ImageCorrelator::instance()->findFeatures(ref_tile, featurePoints);
103  if (featurePoints.empty())
104  return false;
105 
106 #ifdef DEBUG_ATP
107  annotateFeatures(featurePoints);
108 #endif
109 
110  // Loop over features to perform correlations:
111  for (size_t i=0; i<featurePoints.size(); ++i)
112  {
113  string sid = ossimString::toString(tpId).string();
114  ctp.reset(new CorrelationTiePoint(m_refChain, m_cmpChain, featurePoints[i], sid,
115  m_refEllipHgt));
116  ctp->correlate();
117  if (ctp->hasValidPeak())
118  {
119  tileCTPs.push_back(ctp);
120  tpId++;
121  }
122  }
123 
124 #ifdef DEBUG_ATP
125  annotateCorrelations(tileCTPs);
126  writeImage(m_annotatedChain.get(), "tileCTPs_prefilter.tif", "");
127  cout<<"Wrote tileCTPs_prefilter.tif"<<endl;
128 #endif
129 
130  // Loop to eliminate bad peaks and inconsistent CTPs in tile:
131  TiePointList::iterator ctpIter = tileCTPs.begin();
132  int num_pts_affected = 0;
133  while (ctpIter != tileCTPs.end())
134  {
135  shared_ptr<CorrelationTiePoint> ctp ((CorrelationTiePoint*)ctpIter->get());
136  bool peak_changed = ctp->findBestConsistentPeak(tileCTPs);
137  if (peak_changed)
138  {
139  num_pts_affected++;
140  if (!ctp->hasValidPeak())
141  {
142  // Remove this TP from our list as soon as it is discovered that there are no valid
143  // peaks. All TPs in theTpList must have valid peaks:
144  ctpIter = tileCTPs.erase(ctpIter);
145  }
146  else
147  ctpIter++;
148  }
149  else
150  ctpIter++;
151  }
152 
153  // Add remaining tile CTPs to master CTP list and ready for next tile:
154  ctpList.insert(ctpList.end(), tileCTPs.begin(), tileCTPs.end());
155  ref_tile.reset(sequencer->getNextTile().get());
156 
157 #ifdef DEBUG_ATP
158  annotateCorrelations(tileCTPs);
159  writeImage(m_annotatedChain.get(), "tileCTPs_postfilter.tif", "");
160  cout<<"Wrote tileCTPs_postfilter.tif"<<endl;
161 #endif
162 
163  }
164 
165 #ifdef DEBUG_ATP
166  annotateCorrelations(ctpList);
167  writeImage(m_annotatedChain.get(), "annotated.tif", "");
168  cout<<"Wrote annotated.tif"<<endl;
169 #endif
170 
171 
172  return true;
173 }

175 bool CorrelationAtpGenerator::renderTiePointDataAsImages(const ossimFilename& outFile,
176  const ossimFilename& lutFile)
177 {
178  ossimImageSource* chainHead = 0;
179  initialize();
180 
181  // Tile source of correlations and residuals:
182  shared_ptr<ossimCorrelationSource> correlationSource (new ossimCorrelationSource);
183  correlationSource->connectMyInputTo(m_refChain.get());
184  correlationSource->connectMyInputTo(m_cmpChain.get());
185  correlationSource->setCorrStepSize(m_corrStepSize);
186  correlationSource->setRefEllipsoidHgt(m_refEllipHgt);
187  chainHead = correlationSource.get();
188  chainHead->initialize();
189 
190  // Smoothing:
191  shared_ptr<ossimMeanMedianFilter> mmf2 (new ossimMeanMedianFilter());
192  mmf2->connectMyInputTo(chainHead);
193  mmf2->setFilterType(ossimMeanMedianFilter::OSSIM_MEAN_FILL_NULLS);
194  mmf2->setWindowSize(5);
195  chainHead = mmf2.get();
196  chainHead->initialize();
197 
198  // Write a two-band image containing correlations and residuals
199  ossimFilename tmpFile(outFile.fileNoExtension() + "_TMP.tif");
200  writeImage(chainHead, tmpFile, "");
201 
202  // Open the 2-band product image:
203  shared_ptr<ossimImageHandler> handler (
204  ossimImageHandlerRegistry::instance()->open(tmpFile, true, false));
205  if (!handler)
206  return false;
207  chainHead = handler.get(); // Start of new chain here, no resampler needed
208 
209  // Establish the output rect:
210  shared_ptr<ossimImageGeometry> geom (handler->getImageGeometry().get());
211  if (!geom)
212  return false;
213  ossimDrect bounding_rect;
214  geom->getBoundingRect(bounding_rect);
215  m_aoiView = ossimIrect(bounding_rect);
216 
217  shared_ptr<ossimBandSelector> bandSelector (new ossimBandSelector());
218  bandSelector->connectMyInputTo(chainHead);
219  vector<ossim_uint32> bandId(1);
220  bandSelector->enableSource();
221  chainHead = bandSelector.get();
222  chainHead->initialize();
223 
224  // Need to scale results:
225  shared_ptr<ossimLinearStretchRemapper> remapper (new ossimLinearStretchRemapper);
226  remapper->connectMyInputTo(chainHead);
227  double peakThreshold = ImageCorrelator::instance()->getPeakThreshold();
228  double maxResidual = correlationSource->getMaxResidualMagnitude();
229  chainHead = remapper.get();
230  chainHead->initialize();
231 
232  // Scale to 8-bit:
233  shared_ptr<ossimScalarRemapper> scalarRemap (new ossimScalarRemapper);
234  scalarRemap->connectMyInputTo(chainHead);
235  scalarRemap->setOutputScalarType(OSSIM_UINT8);
236  chainHead = scalarRemap.get();
237  chainHead->initialize();
238 
239  if (!lutFile.empty())
240  {
241  shared_ptr<ossimIndexToRgbLutFilter> lut (new ossimIndexToRgbLutFilter());
242  lut->connectMyInputTo(chainHead);
243  lut->setLut(lutFile);
244  chainHead = lut.get();
245  chainHead->initialize();
246  }
247 
248  // Correlation file output:
249  bandId[0] = 0;
250  bandSelector->setOutputBandList(bandId);
251  remapper->setMinPixelValue(peakThreshold, 0);
252  remapper->setMaxPixelValue(1.0, 0);
253  ossimFilename corrFile(outFile.fileNoExtension() + "_COR.tif");
254  writeImage(chainHead, corrFile, lutFile);
255 
256  // Residual file output:
257  bandId[0] = 1;
258  bandSelector->setOutputBandList(bandId);
259  remapper->setMinPixelValue(0.0, 0);
260  remapper->setMaxPixelValue(maxResidual, 0);
261  ossimFilename resFile(outFile.fileNoExtension() + "_RES.tif");
262  writeImage(chainHead, resFile, lutFile);
263 
264  // Remove the intermediate 2-band file
265  ossimFilename::remove(tmpFile);
266  return true;
267 }
