63 void Image::loadJSON(const Json::Value& json_node)
64 {
65  ostringstream xmsg;
66  xmsg<<__FILE__<<": loadJSON(JSON) -- ";
67 
68  // Parse JSON. Filename is required:
69  if (json_node.isMember("filename"))
70  {
71  m_filename = json_node["filename"].asString();
72  }
73  else
74  {
75  xmsg<<"JSON node missing required field: \"filename\".";
76  throw ossimException(xmsg.str());
77  }
78 
79  // Entry index defaults to 0 if not present:
80  if (json_node["entryIndex"].isUInt())
81  m_entryIndex = json_node["entryIndex"].asUInt();
82 
83  // Sensor model defaults to most accurate available if not provided (indicated by blank name):
84  if (json_node.isMember("sensorModel"))
85  m_modelName = json_node["sensorModel"].asString();
86 
87  // Establish the sensor model. This also sets the official image ID, which will be overwritten
88  // if JSON field provided
89  string modelState = json_node["modelState"].asString();
90 #if OSSIM_HAS_MSP
91  if (modelState.size())
92  {
93  MSP::SMS::SensorModelService sms;
94  csm::Model* base = sms.createModelFromState(modelState.c_str());
95  m_csmModel = dynamic_cast<csm::RasterGM*>(base);
96  m_imageId = m_csmModel->getImageIdentifier();
97  }
98  else
99  {
100  getCsmSensorModel();
101  }
102 #else
103  ossimKeywordlist kwl;
104  kwl.parseString(modelState);
105  shared_ptr<ossimProjection> proj (
106  ossimProjectionFactoryRegistry::instance()->createProjection(kwl));
107  m_sensorModel = dynamic_pointer_cast<ossimSensorModel>(proj);
108 #endif
109 }
