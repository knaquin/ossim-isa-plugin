128 bool ossimIsaTool::execute()
129 {
130  ostringstream xmsg;
131 
132  // Read input:
133  stringstream json_query;
134  streambuf *rqBuf = json_query.rdbuf();
135  (*m_inputStream) >> rqBuf;
136  try
137  {
138  Json::Value queryRoot;
139  Json::Reader reader;
140  bool parsingSuccessful = reader.parse( json_query.str(), queryRoot );
141  if ( !parsingSuccessful )
142  {
143  xmsg<<"Failed to parse JSON query string. \n"
144  << reader.getFormattedErrorMessages()<<endl;
145  throw ossimException(xmsg.str());
146  }
147 
148  // Fetch service name and route accordingly:
149  string serviceName = queryRoot["service"].asString();
150  shared_ptr<ServiceBase> service;
151  if (serviceName.compare("getVersion") == 0)
152  service.reset(new VersionService);
153  else if (serviceName.compare("sensorModels") == 0)
154  service.reset(new SensorModelService);
155  else if (serviceName.compare("autoTiePoint") == 0)
156  service.reset(new AutoTiePointService);
157 
158 #if OSSIM_HAS_MSP
159  else if (serviceName.compare("sourceSelection") == 0)
160  service.reset(new SourceSelectionService);
161  else if (serviceName.compare("triangulation") == 0)
162  service.reset(new TriangulationService);
163  // else if (serviceName.compare("mensuration") == 0)
164  // service.reset(new RegistrationService);
165 #endif
166  else
167  {
168  xmsg<<"Wrong service <"<<serviceName<<"> requested."<<endl;
169  throw ossimException(xmsg.str());
170  }
171 
172  // Call actual service and get response:
173  service->loadJSON(queryRoot);
174  service->execute();
175  Json::Value response;
176  service->saveJSON(response);
177 
178  // Serialize JSON object for return:
179  Json::StyledWriter writer;
180  string json_response = writer.write(response);
181  (*m_outputStream) << json_response;
182  }
183  catch(ossimException &e)
184  {
185  cerr<<"Exception: "<<e.what()<<endl;
186  *m_outputStream<<"{ \"ERROR\": \"" << e.what() << "\" }\n"<<endl;
187  }
188 
189  // close any open file streams:
190  ifstream* si = dynamic_cast<ifstream*>(m_inputStream);
191  if (si)
192  {
193  si->close();
194  delete si;
195  }
196  ofstream* so = dynamic_cast<ofstream*>(m_outputStream);
197  if (so)
198  {
199  so->close();
200  delete so;
201  }
202 
203  return true;
204 }
