59  // Establish the nominal height above ellipsoid for MSL at this location:

96 shared_ptr<ossimImageChain> AtpGeneratorBase::constructChain(const ossimFilename& imageFile)
97 {
98  ossimImageHandlerRegistry* factory = ossimImageHandlerRegistry::instance();
99  shared_ptr<ossimImageHandler> handler (factory->open(imageFile));
100  if (!handler)
101  return NULL;
102 
103  shared_ptr<ossimImageChain> chain (new ossimImageChain);
104  chain->add(handler.get());
105 
106  // Performs weighted combining of bands (no ratios here):
107  unsigned int numBands = handler->getNumberOfOutputBands();
108  if (numBands > 1)
109  {
110  shared_ptr<ossimBandAverageFilter> band_combiner (
111  new ossimBandAverageFilter(handler.get(), m_bandWeights));
112  chain->add(band_combiner.get());
113  }
114 
115  // IVT and Renderer. The view geometry is set to the tie-point and GSD of the first (REF) image:
116  shared_ptr<ossimImageRenderer> renderer (new ossimImageRenderer);
117  chain->add(renderer.get());
118  shared_ptr<ossimImageGeometry> image_geom (handler->getImageGeometry().get());
119  if (!m_viewGeom)
120  {
121  ossimDpt gsd(image_geom->getMetersPerPixel());
122  ossimGpt proj_origin;
123  image_geom->getTiePoint(proj_origin, false);
124  shared_ptr<ossimUtmProjection> proj (new ossimUtmProjection);
125  proj->setOrigin(proj_origin);
126  proj->setMetersPerPixel(gsd);
127  proj->setUlTiePoints(proj_origin);
128 
129  m_viewGeom.reset(new ossimImageGeometry(NULL, proj.get()));
130  }
131  shared_ptr<ossimImageViewProjectionTransform> ivt (
132  new ossimImageViewProjectionTransform(image_geom.get(), m_viewGeom.get()));
133  renderer->setImageViewTransform(ivt.get());
134 
135  // Add cache after resampler:
136  shared_ptr<ossimCacheTileSource> cache (new ossimCacheTileSource());

209 {

223  ossimIpt lr (featurePoints[i].x + halfWidth, featurePoints[i].y + halfWidth);
224  ossimIpt ll (featurePoints[i].x - halfWidth, featurePoints[i].y + halfWidth);
225

227  ossimAnnotationLineObject* line2 = new ossimAnnotationLineObject(ur, lr, 255,255,0);
