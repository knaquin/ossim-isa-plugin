215 bool ImageCorrelator::OpenCVCorrelation(const shared_ptr<ossimImageData> cmppatch,
216  const ossimIpt& cmpPatchLocV,
217  const shared_ptr<ossimImageData> refpatch,
218  const ossimIpt& refPatchLocV,
219  CorrelationTiePoint* correlationTP)
220 {
221  static const char* MODULE = "ImageCorrelator::OpenCVCorrelation() -- ";
222 
223  int ref_patch_size = refpatch->getWidth();
224  int refPatchOffset = ref_patch_size / 2; // assumes square patch
225 
226  IplImage *cmpimage, *refimage, *result;
227  int method;
228 
229  // Convert image data to CV IPL form:
230  cmpimage = convertToIpl32(cmppatch);
231  refimage = convertToIpl32(refpatch);
232  if (!cmpimage || !refimage)
233  {
234  cout << MODULE << "Error encountered creating IPL image tiles. Aborting..." << endl;
235  return false;
236  }
237 
238  // Assign the peak threshold depending on whether it is same collect or diff collect:
239  double peak_threshold = m_peakThreshold;
240 
241  CvSize resdim;
242  resdim.height = cmpimage->height - refimage->height + 1;
243  resdim.width = cmpimage->width - refimage->width + 1;
244  result = cvCreateImage(resdim, IPL_DEPTH_32F, 1);
245  if (!result)
246  {
247  cout << MODULE << "Error encountered creating IPL image tile (result). Aborting..." << endl;
248  return false;
249  }
250  switch (m_method)
251  {
252  case CV_CORRCOEF:
253  method = CV_TM_CCOEFF;
254  break;
255  case CV_CROSSCORR:
256  method = CV_TM_CCORR;
257  break;
258  case CV_CROSSCORR_NORM:
259  method = CV_TM_CCORR_NORMED;
260  break;
261  case CV_SQUAREDDIFF:
262  method = CV_TM_SQDIFF;
263  break;
264  case CV_SQUAREDDIFF_NORM:
265  method = CV_TM_SQDIFF_NORMED;
266  break;
267  case CV_CORRCOEF_NORM:
268  default:
269  method = CV_TM_CCOEFF_NORMED;
270  break;
271  }
272 
273  // Do the actual correlation.
274  cvMatchTemplate(cmpimage, refimage, result, method);
275 
276  ossimIpt cmpPeakLocV;
277  float corrCoef;
278 
279  // Search through all the peaks and the tiepoint object will track which is best.
280  multimap<float /* corr_coef */, ossimIpt /* loc */, greater<float> > peaks_map;
281  for (int dy = 0; dy < result->height; dy++)
282  {
283  for (int dx = 0; dx < result->width; dx++)
284  {
285  // Fetch the pixel's corr coef and insert in map to order by strongest peak:
286  corrCoef = ((float*) (result->imageData + result->widthStep * dy))[dx];
287  cmpPeakLocV.x = cmpPatchLocV.x + refPatchOffset + dx;
288  cmpPeakLocV.y = cmpPatchLocV.y + refPatchOffset + dy;
289 
290  if (corrCoef >= peak_threshold)
291  peaks_map.insert(pair<float, ossimIpt>(corrCoef, cmpPeakLocV));
292  }
293  }
294 
295  if (peaks_map.size() > 0)
296  {
297  // The peaks are in order by greatest correlation value. Send the best to the CTP:
298  multimap<float, ossimIpt, greater<float> >::iterator peak = peaks_map.begin();
299  for (unsigned int i = 0; (i < m_maxNumPeaks) && (peak != peaks_map.end()); i++)
300  {
301  cmpPeakLocV = peak->second;
302  corrCoef = peak->first;
303  correlationTP->addPeak(cmpPeakLocV, corrCoef);
304  peak++;
305  }
306  }
307 
308  cvReleaseImage(&result);
309  cvReleaseImage(&refimage);
310  cvReleaseImage(&cmpimage);
311 
312  return true;

315 void ImageCorrelator::findFeatures(const shared_ptr<ossimImageData> imageChip,
316  vector<ossimIpt>& feature_points)
317 {
318  const char* MODULE = "ImageCorrelator::findFeatures(): ";
319 
320  feature_points.clear();
321  if (!imageChip || (imageChip->getDataObjectStatus() == OSSIM_EMPTY))
322  {
323  cout<<MODULE<<"WARNING: NULL or empty image data passed in."<<endl;
324  return;
325  }
326 
327  // THE FOLLOWING ARE CRITICAL VALUES AND MAYBE SHOULD BE READ FROM KWL:
328  double refPatchSize = CorrelationTiePoint::getRefPatchSize(); // min distance between points
329  static const double quality = 0.001; // minimum strength of feature considered
330  static const int use_harris=1;
331  static const double harris_k=0.04;
332  static const int averaging_block_size = 5;
333 
334  IplImage *ipl_image = convertToIpl(imageChip);
335  if(!ipl_image)
336  {
337  cout<<MODULE << "ERROR: Got null IplImage pointer converting from ossimImageData*!";
338  return;
339  }
340 
341  // Need to mask pixels outside of intersection:
342  ossimIpt tileUL (imageChip->getImageRectangle().ul());
343  IplImage* mask = 0;
344  double null_pixel = imageChip->getNullPix(0);
345  if (imageChip->getDataObjectStatus() == OSSIM_PARTIAL)
346  {
347  CvSize bufsize = cvGetSize(ipl_image);
348  mask = cvCreateImage(bufsize, IPL_DEPTH_8U, 1 );
349  ossimDpt pt;
350  long offset = 0;
351  for (int y=0; y<bufsize.height; y++)
352  {
353  for (int x=0; x<=bufsize.width-1; x++)
354  {
355  if (imageChip->getPix(offset) != null_pixel)
356  mask->imageData[offset] = (unsigned char) 0xFF;
357  else
358  mask->imageData[offset] = (unsigned char) 0;
359  ++offset;
360  }
361  }
362  }
363 
364  //##### DEBUG: Show feature search patch ###
365  //cvNamedWindow("ref_image", 1); //###
366  //cvShowImage("ref_image", ipl_image); //###
367 
368  // Use CV to find features in the patch:
369  IplImage* eig = cvCreateImage(cvGetSize(ipl_image), 32, 1 );
370  IplImage* temp = cvCreateImage(cvGetSize(ipl_image), 32, 1 );
371 
372  // Adjust the number of features to look for if we are doing adaptive correlation thresholding
373  // (ACT). This would be more appropriate to call "secondary feature consideration":
374  int num_pts = m_numFeaturesPerTile;
375 
376  CvPoint2D32f* points = (CvPoint2D32f*)cvAlloc(num_pts*sizeof(CvPoint2D32f));
377  cvGoodFeaturesToTrack(ipl_image, eig, temp, points, &num_pts, quality,
378  refPatchSize, mask, averaging_block_size, use_harris, harris_k);
379 
380  // Loop over all features found in this patch and assign a feature point at that view
381  // point. IMPORTANT NOTE: it is assumed that the ordering of the points[] array is by
382  // descending strength of feature.
383  unsigned int num_added = 0;
384  for(int i=0; i<num_pts; i++)
385  {
386  // Correct point for the view-space offset of the search patch since points are relative to
387  // patch upper left:
388  CvPoint pt=cvPointFrom32f(points[i]);
389  ossimDpt viewPt (pt.x, pt.y);
390  viewPt += tileUL;
391 
392  // Good feature found, set the feature point's attributes and save to list:
393  feature_points.push_back(viewPt);
394  ++num_added;
395  }
396 
397  // Cleanup mem use:
398  cvReleaseImage(&eig);
399  cvReleaseImage(&temp);
400  cvReleaseImage(&mask);
401  cvReleaseImage(&ipl_image);
402  cvFree((void**)&points);
403 
404  return;
405 }
