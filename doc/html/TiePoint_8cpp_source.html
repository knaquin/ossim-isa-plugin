111 void TiePoint::loadJSON(const Json::Value& json_node)
112 {
113  ostringstream xmsg;
114  xmsg<<__FILE__<<": TiePoint(JSON) --";
115 
116  // ID
117  setTiePointId(json_node["id"].asString());
118 
119  // Type
120  string tpType = json_node["type"].asString();
121  switch (tpType[0])
122  {
123  case 'M':
124  m_type = TiePoint::MANUAL;
125  break;
126  case 'A':
127  m_type = TiePoint::AUTO;
128  break;
129  case 'G':
130  m_type = TiePoint::GCP;
131  break;
132  default:
133  xmsg<<"Tiepoint JSON field \"type\" must be specified as 'M', 'A', or 'G'.";
134  throw ossimException(xmsg.str());
135  }
136 
137  // Read the GCP ID if present:
138  if (m_type == TiePoint::GCP)
139  m_gcpId = json_node["gcpId"].asString();
140 
141  // Image points
142  const Json::Value& imagePoints = json_node["imagePoints"];
143  if (!imagePoints || (imagePoints.size() < 1))
144  {
145  xmsg<<"Tiepoint JSON field \"imagePoints\" not found or is empty!";
146  throw ossimException(xmsg.str());
147  }
148 
149  // Loop over points on each image:
150  for (int i=0; i<imagePoints.size(); ++i)
151  {
152  const Json::Value& p = imagePoints[i];
153  if (!p || !(p["imageId"].isString()) || !(p["x"]) || !(p["y"]) || (p["covariance"].size()!=3))
154  {
155  xmsg<<"Tiepoint JSON field \"imagePoints\" entry is ill-formed or not complete:\n"
156  <<p.toStyledString()<<endl;
157  throw ossimException(xmsg.str());
158  }
159 
160  string imageId = p["imageId"].asString();
161  m_imageIds.push_back(imageId);
162 
163  ossimDpt xy(p["x"].asDouble(), p["y"].asDouble());
164  m_imagePoints.push_back(xy);
165 
166  const Json::Value& covariance = p["covariance"];
167  NEWMAT::SymmetricMatrix c (2);
168  c(1,1) = covariance[0].asDouble();
169  c(2,2) = covariance[1].asDouble();
170  c(1,2) = covariance[2].asDouble();
171  m_covariances.push_back(c);
172  }

175 void TiePoint::saveJSON(Json::Value& json_node) const
176 {
177  // ID

179 
180  // Type

196 
197  // Image points

199  // Loop over points on each image:

201  {
202  jsonList[i]["imageId"] = m_imageIds[i];
203  jsonList[i]["x"] = m_imagePoints[i].x;
204  jsonList[i]["y"] = m_imagePoints[i].y;
205 
206  Json::Value covJson (Json::arrayValue);
207  covJson[0] = m_covariances[i](1,1);
208  covJson[1] = m_covariances[i](2,2);
209  covJson[2] = m_covariances[i](1,2);
210  jsonList[i]["covariance"] = covJson;
211  }
212  json_node["imagePoints"] = jsonList;
