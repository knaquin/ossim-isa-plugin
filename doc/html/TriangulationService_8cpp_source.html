110 void TriangulationService::fillGcpList(MSP::GroundPointList& mspGroundPts)
111 {
112  MSP::Matrix mspCov(3,3);
113  NEWMAT::SymmetricMatrix ossimCov;
114  mspCov.zero();
115  ossimEcefPoint ecf;
116  const vector< shared_ptr<GroundControlPoint> >& gcpList = m_photoBlock->getGroundPointList();
117  for (int g=0; g<gcpList.size(); ++g)
118  {
119  ecf = gcpList[g]->getECF();
120  ossimCov = gcpList[g]->getCovariance();
121  mspCov.setElement(0, 0, ossimCov(1,1));
122  mspCov.setElement(1, 1, ossimCov(2,2));
123  mspCov.setElement(2, 2, ossimCov(3,3));
124  mspCov.setElement(0, 1, ossimCov(1,2));
125  mspCov.setElement(0, 2, ossimCov(1,3));
126  mspCov.setElement(1, 2, ossimCov(2,3));
127  mspCov.setElement(1, 0, ossimCov(2,1));
128  mspCov.setElement(2, 0, ossimCov(3,1));
129  mspCov.setElement(2, 1, ossimCov(3,2));
130  MSP::GroundPoint mspGpt (ecf.x(), ecf.y(), ecf.z(), gcpList[g]->getId());
131  mspGpt.setCovariance(mspCov);
132  mspGroundPts.push_back(mspGpt);
133  }
134 }

136 void TriangulationService::fillTpList(MSP::ImagePointList& mspImagePts)
137 {
138  MSP::Matrix mspCov(2,2);
139  NEWMAT::SymmetricMatrix ossimCov;
140  mspCov.zero(); // off-diagonals zeroed
141  string imageId;
142  string pointId;
143  unsigned int imgCount=0;
144  ossimDpt imagePt;
145  const std::vector<std::shared_ptr<TiePoint> >& tiePoints =
146  m_photoBlock->getTiePointList();
147  for (int t=0; t<tiePoints.size(); ++t)
148  {
149  // Fetch the photoblock's entry for this tiepoint:
150  pointId = tiePoints[t]->getGcpId();
151  if (pointId.empty())
152  pointId = tiePoints[t]->getTiePointId();
153  imgCount = tiePoints[t]->getImageCount();
154  for (int i=0; i<imgCount; ++i)
155  {
156  tiePoints[t]->getImagePoint(i, imageId, imagePt, ossimCov);
157  mspCov.setElement(0, 0, ossimCov(2,2)); // flip for x,y versus line sample
158  mspCov.setElement(1, 1, ossimCov(1,1));
159  MSP::ImagePoint mspImagePoint(imagePt.line, imagePt.samp, imageId, pointId, mspCov);
160  //cout<<"mspImagePoint:\n"<<mspImagePoint.toString()<<endl;//###
161  mspImagePts.push_back( mspImagePoint );
162  }
163  }
164 }
