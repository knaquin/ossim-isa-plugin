174 bool AutoTiePoint::checkConsistency(const TiePointList& ctpList)
175 {
176  ConsistencyTestInfo* cti = ConsistencyTestInfo::instance();
177  ossimDpt R1;
178  unsigned int num_consistent_neighbors = 0;
179 
180  // Loop over all of the TPs neighbors:
181  TiePointList::const_iterator neighbor;
182  for(neighbor=ctpList.begin();neighbor!=ctpList.end();++neighbor)
183  {
184  shared_ptr<AutoTiePoint> atp (dynamic_pointer_cast<AutoTiePoint>(*neighbor));
185  if (!atp || (atp.get() == this))
186  continue;
187 
188  // APC is not enabled, so simply check against neighbor's active (primary) peak:
189  R1 = atp->m_residual;
190  if (residualsAreConsistent(R1))
191  num_consistent_neighbors++;
192  }
193 
194  // Decide if this peak was good or out there:
195  unsigned int n = cti->m_numConsistentThreshold;
196  unsigned int p = (unsigned int) ceil(cti->m_percentConsistentThreshold*(ctpList.size()-1));
197  if (p == 0)
198  p = 1;
199  unsigned int min_num_consistent_neighbors = (n < p ? n : p);
200  if (min_num_consistent_neighbors == 0)
201  min_num_consistent_neighbors = p;
202  if (num_consistent_neighbors < min_num_consistent_neighbors)
203  return false;
204 
205  return true;
206 }

208 bool AutoTiePoint::residualsAreConsistent(const ossimDpt& R1)
209 {
210  static const double MAGNITUDE_DIFF_TOLERANCE = 3.0; // For small residuals to avoid pruning due to dir
211 
212  ConsistencyTestInfo* cti = ConsistencyTestInfo::instance();
213  double cos_theta, mag_ratio;
214 
215  // Compute angle and magnitude ratio between residuals. Note no check for valid peak:
216  double rr0 = m_residual.x*m_residual.x + m_residual.y*m_residual.y;
217  double rr1 = R1.x*R1.x + R1.y*R1.y;
218  if (rr1 == 0.0)
219  {
220  cos_theta = 0;
221  if (rr0 == 0.0)
222  mag_ratio = 1.0;
223  else
224  mag_ratio = OSSIM_DEFAULT_MAX_PIX_DOUBLE;
225  }
226  else if (rr0 == 0.0)
227  {
228  cos_theta = 0;
229  if (rr1 == 0.0)
230  mag_ratio = OSSIM_DEFAULT_MAX_PIX_DOUBLE;
231  else
232  mag_ratio = 1.0;
233  }
234  else
235  {
236  mag_ratio = rr0/rr1;
237  cos_theta = (m_residual.x*R1.x + m_residual.y*R1.y)/sqrt(rr0*rr1);
238  }
239 
240  // Test for consistency with this neighbor's peak:
241  double diff = (m_residual-R1).length();
242  if ((diff < MAGNITUDE_DIFF_TOLERANCE) ||
243  ((mag_ratio < cti->m_magnitudeSquaredThreshold) &&
244  (mag_ratio > cti->m_inverseMag2Threshold) &&
245  (cos_theta > cti->m_cosAngleThreshold))) return true;
246 
247  return false;
248 }
