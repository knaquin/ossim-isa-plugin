71 ossimRefPtr<ossimImageData> ossimCorrelationSource::getTile(const ossimIrect& tileRect,
72  ossim_uint32 resLevel)
73 {
74  if(!m_tile.get())
75  {
76  // try to initialize
77  initialize();
78  if(!m_tile.get())
79  return m_tile;
80  }
81 
82  if(getNumberOfInputs() < 2)
83  return 0;
84 
85  m_tile->setImageRectangle(tileRect);
86  m_tile->makeBlank();
87 
88  shared_ptr<ossimImageSource> ref_source ((ossimImageSource*)getInput(0));
89  shared_ptr<ossimImageSource> cmp_source ((ossimImageSource*)getInput(1));
90  int tpId = 0;
91 
92  if (m_featureMode)
93  {
94  // Need to search for features in the REF tile first, then consider only those locations
95  // for correlating:
96  ossimRefPtr<ossimImageData> ref_tile = ref_source->getTile(tileRect, resLevel);
97  if (ref_tile->getDataObjectStatus() == OSSIM_EMPTY)
98  return m_tile;
99  vector<ossimIpt> featurePoints;
100  shared_ptr<ossimImageData> tileptr (ref_tile.get());
101  ImageCorrelator::instance()->findFeatures(tileptr, featurePoints);
102  if (featurePoints.empty())
103  return m_tile;
104 
105  // Loop over features to perform correlations:
106  for (size_t i=0; i<featurePoints.size(); ++i)
107  {
108  string sid = ossimString::toString(tpId).string();
109  shared_ptr<CorrelationTiePoint> ctp (
110  new CorrelationTiePoint(ref_source, cmp_source, featurePoints[i],
111  sid, m_refEllipHgt));
112  if (correlate(ctp, tileRect))
113  tpId++;
114  }
115  }
116  else
117  {
118  // Scan raster-fashion along REF tile and correlate at intervals:
119  for (int y = tileRect.ul().y; y < tileRect.lr().y; y += m_corrStepSize)
120  {
121  for (int x = tileRect.ul().x; x < tileRect.lr().x; x += m_corrStepSize)
122  {
123  string sid = ossimString::toString(tpId).string();
124  ossimDpt viewPt(x, y);
125  shared_ptr<CorrelationTiePoint> ctp (
126  new CorrelationTiePoint(ref_source, cmp_source, viewPt, sid, m_refEllipHgt));
127  if (correlate(ctp, tileRect))
128  tpId++;
129  }
130  }
131  }
132  m_tile->validate();
133  //m_tile->write("chip.ras");
134  return m_tile;
135 }

137 bool ossimCorrelationSource::correlate(shared_ptr<CorrelationTiePoint> ctp, const ossimIrect& tileRect )
138 {
139  bool found = false;
140 
141  m_tile->getImageRectangle();
142  ossimDpt residual;
143  double corr_val, res_mag;
144  ossimIpt chipLoc;
145  double** destBand = new double*[2];
146  destBand[0] = (double*)m_tile->getBuf(0);
147  destBand[1] = (double*)m_tile->getBuf(1);
148  long offset = 0;
149 
150  int halfStep = (int) m_corrStepSize / 2;
151  ctp->setCmpPatchSize(m_nominalCmpPatchSize);
152  ctp->correlate();
153  if (ctp->hasValidPeak())
154  {
155  found = true;
156  m_tiePoints.push_back(ctp);
157  ctp->getVectorResidual(residual);
158  ctp->getCorrelationValue(corr_val);
159  res_mag = residual.length();
160  ossimDpt viewPt;
161  ctp->getRefViewPoint(viewPt);
162 
163  // Fill an area in the destination buffer equal to the corr step size:
164  for (chipLoc.y = viewPt.y-halfStep; chipLoc.y <= viewPt.y+halfStep; ++chipLoc.y)
165  {
166  for (chipLoc.x = viewPt.x-halfStep; chipLoc.x <= viewPt.x+halfStep; ++chipLoc.x)
167  {
168  if (tileRect.pointWithin(chipLoc))
169  {
170  offset = (chipLoc.y-tileRect.ul().y)*tileRect.width()+chipLoc.x-tileRect.ul().x;
171  destBand[0][offset] = corr_val;
172  destBand[1][offset] = res_mag;
173  if (res_mag > m_maxResidualMagnitude)
174  m_maxResidualMagnitude = res_mag;
175  }
176  else
177  cout << "\nossimCorrelationSource::correlate() Found point outside tile rect!"<<endl;
178  }
179  }
180  }
181  return found;
182 }
183 }
