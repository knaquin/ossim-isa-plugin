53 void SourceSelectionService::execute()
54 {
55  ostringstream xmsg;
56  xmsg<<"SourceSelectionService::execute(): ";
57 
58  m_estimatedCE90 = -1;
59  m_estimatedLE90 = -1;
60 
61  size_t ncands = m_candidateImages.size();
62  if ((ncands == 0) || (m_mustUse.size() != ncands))
63  return;
64 
65  // Need the reference ground point in ECEF coordinates:
66  double x, y, z;
67  csm::EcefCoord csmEcefPt (m_refPt.x(), m_refPt.y(), m_refPt.z());
68 
69  // Prepare MSP data structures:
70  MSP::SS::CandidateImageList mspCandidates;
71  MSP::GroundPointList mspGcpList; // Left empty
72  MSP::JointCovMatrixList mspJcmList; // TODO: Left empty -- no way to model cross-correlation
73 
74  // Start a new session NOW:
75  m_session = SessionManager::newSession();
76 
77  // Loop to add all images:
78  for (size_t i=0; i<ncands; ++i)
79  {
80  shared_ptr<Image> image (m_candidateImages[i]);
81  const char* fname = image->getFilename().c_str();
82  const char* modelName =image->getModelName().c_str();
83  const csm::RasterGM* model = image->getCsmSensorModel();
84  if (!model)
85  {
86  xmsg<<"Could not instantiate sensor model from image file: <"<<fname<<">."<<endl;
87  throw ossimException(xmsg.str());
88  }
89 
90  // Construct MSP candidate image representing this file:
91  MSP::SS::CandidateImage::Usage usage_req = MSP::SS::CandidateImage::CAN_USE;
92  if (m_mustUse[i])
93  usage_req = MSP::SS::CandidateImage::MUST_USE;
94  mspCandidates.push_back(MSP::SS::CandidateImage(model, usage_req));
95  }
96 
97  // Find best subset of images:
98  MSP::SS::SourceSelectionCriteria mspCriteria;
99  mspCriteria.setMinImages(2);
100  mspCriteria.setDesiredErrorProp(m_desiredCE90, m_desiredLE90);
101  MSP::SS::SourceSelectionService mspSSS;
102  MSP::SS::SourceSelectionResult mspAbsResult;
103  mspSSS.sourceSelect(mspCandidates, mspJcmList, mspCriteria, csmEcefPt, mspAbsResult);
104 
105  // A new photoblock is initialized to represent images and eventually tiepoints and GCPs used
106  // in this session. The candidate images will be added to it:
107  shared_ptr<PhotoBlock> photoblock (m_session->getPhotoBlock());
108 
109  // Process result. Need to correlate instance of sensor model with corresponding filename:
110  m_meetsCriteria = mspAbsResult.meetsCriteria();
111  const MSP::CsmSensorModelList& subsetModels = mspAbsResult.getSensorModelList();
112  for (int i=0; i<subsetModels.size(); ++i)
113  {
114  string imageID1 = subsetModels[i]->getImageIdentifier();
115  for (int j=0; j<m_candidateImages.size(); ++j)
116  {
117  string imageID2 = m_candidateImages[j]->getImageId();
118  if (imageID1.compare(imageID2) == 0)
119  {
120  // found match. Need to instantiate the Image object and save to list:
121 #if DEBUG_ON
122  cout<<"Image ID match: "<<imageID1<<endl;
123 #endif
124  photoblock->addImage(m_candidateImages[j]);
125  break;
126  }
127  }
128  }
129 
130  // Need to compute CE/LE given covariance:
131  computeAccuracy(mspAbsResult);
132 
133  // Finally, save it to the Db:
134  //m_session->saveState();
135 
136  return;

181 void SourceSelectionService::computeAccuracy(MSP::SS::SourceSelectionResult& mspAbsResult)
182 {
183  const MSP::Matrix& cov = mspAbsResult.getEnuCovariance();
184 
185  // CE and LE need to be computed from the 3x3 ENU covaiance returned because MSP does not make
186  // those values available. Take horizontal EC 2x2, and determine eigenvalues to be max and min
187  // error ellipse axes, then compute CE90 given min/max ratio. Use single U variance for LE90:
188  double a = 0.5 * (cov[0][0] + cov[1][1]);
189  double b2 = (cov[0][0]-cov[1][1])*(cov[0][0]-cov[1][1]) + 4*cov[0][1]*cov[0][1];
190  double b = 0.5 * sqrt(b2);
191  double su = sqrt(a + b);
192  double sv = sqrt(a - b);
193  double c = su/sv;
194  m_estimatedCE90 = (1.6545 - 0.13913*c + 0.6324*c*c) * su;
195  m_estimatedLE90 = 1.6449 * sqrt(cov[2][2]);
196 
197 #if DEBUG_ON
198  // Report results to stdout:
199  cout << "\nResults from MSP sourceSelect(): ";
200  cout << mspAbsResult.toString() << endl << endl;
201  cout<<"ENU Covariance: "
202  <<"\n\t"<<cov[0][0]<<" "<<cov[0][1]<<" "<<cov[0][2]
203  <<"\n\t"<<cov[1][0]<<" "<<cov[1][1]<<" "<<cov[1][2]
204  <<"\n\t"<<cov[2][0]<<" "<<cov[2][1]<<" "<<cov[2][2]<<"\n"<<endl;
205  const std::vector<std::shared_ptr<Image> >& bestSubset =
206  m_session->getPhotoBlock()->getImageList();
207  cout << "Results from 3DISA sourceSelect(): \n Image File Selection:"<<endl;
208  for (int i=0; i<bestSubset.size(); ++i)
209  {
210  cout<<" "<<bestSubset[i]<<endl;
211  }
212  cout<<"\n Computed CE90: "<<m_estimatedCE90<<endl;
213  cout<<" Computed LE90: "<<m_estimatedLE90<<endl;
214  cout<<" meets criteria: "<<m_meetsCriteria<<"\n"<<endl;
215 #endif
216 }
