89 {
90  // Establish the REF patch:
91  ref_patch = ossimIrect(m_refViewPt, m_refPatchSize, m_refPatchSize);
92 
93  if (m_cmpPatchSize == 0)
94  {
95  // Fetch the CE90 of both images to determine cmp image search patch size:
96  ossimSensorModel* ref_sm = dynamic_cast<ossimSensorModel*>(
97  m_refIVT->getImageGeometry()->getProjection());

99  m_cmpIVT->getImageGeometry()->getProjection());
100  if (ref_sm && cmp_sm)
101  {
102  double ref_ce = ref_sm->getNominalPosError();
103  double cmp_ce = cmp_sm->getNominalPosError();
104  double total_error = ref_ce + cmp_ce; // meters -- being conservative here and not doing rss
105  ossimDpt gsd = m_refIVT->getOutputMetersPerPixel();
106  double nominalGsd = (gsd.x + gsd.y)/2.0;
107  double searchSizePixels = 3.0*total_error/nominalGsd; // should be 2.0 factor, but give a bit more
108  m_cmpPatchSize = m_refPatchSize + searchSizePixels;
109  }

114  }
115  }
116  cmp_patch = ossimIrect(m_refViewPt, m_cmpPatchSize, m_cmpPatchSize);
117 
118  return true;

142 bool CorrelationTiePoint::findBestConsistentPeak(TiePointList& ctpList)
143 {
144  bool changed = false;
145 
146  unsigned int num_neighbors = ctpList.size()-1;
147  if (num_neighbors <= 0)
148  {
149  ctpList.clear();
150  return false;
151  }
152 
153  while (m_correlationPeaks.size())
154  {
155  // Check if this TP peak is consistent with at least one nearby TP:
156  // All the work is done by checkConsistency(). Returns TRUE if consistent:
157  if (checkConsistency(ctpList))
158  break;
159 
160  // Bad peak detected, remove it from the list so that next one becomes active:
161  //SMART_NOTIFY("Removing inconsistent peak for TP #"<<theTpId);
162  m_correlationPeaks.erase(m_correlationPeaks.begin());
163  changed=true;
164  if (m_correlationPeaks.size() != 0)
165  recomputeResidual();
166  }
167 
168  // We are required to use all neighbors available for consistency check, even those considered
169  // invalid by prior consistency checks. This prevents lone wildpoints from surviving the prune
170  // due to surrounding wildpoints being eliminated first. However, now that we are done referencing
171  // these neighbors, we can prune the list down to consider only active TPs in the next consistency
172  // check for this TP:
173  TiePointList::iterator neighbor = ctpList.begin();
174  while (neighbor != ctpList.end())
175  {
176  shared_ptr<CorrelationTiePoint> ctp (dynamic_pointer_cast<CorrelationTiePoint>(*neighbor));
177  if (!ctp || (ctp.get() == this))
178  continue;
179  if (!ctp->hasValidPeak())
180  ctpList.erase(neighbor++);
181  else
182  ++neighbor;
183  }
184 
185  return changed;
186 }
