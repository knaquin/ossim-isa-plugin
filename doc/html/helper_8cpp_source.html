182 bool ce81Compliant(Metadata metaHost, Metadata metaQry, const char* hostImage,
183  const char* queryImage, ImagePoint a, ImagePoint b, double& error)
184 {
185  ImagePoint pointOnQry = pointToPoint(a, hostImage, queryImage);
186  double distance = sqrt(
187  pow(((pointOnQry.x - b.x) * metaQry.meters_per_pixel_x), 2) + pow(
188  ((pointOnQry.y - b.y) * metaQry.meters_per_pixel_y), 2));
189  error = distance;
190  return (distance <= (metaHost.ce90 + metaQry.ce90));
